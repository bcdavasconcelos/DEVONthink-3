# theLabel

```applescript
property theProperty : "label"

-- Labels

-- No label
set theRecords to search theProperty & "==0"		
-- Label: To do
set theRecords to search theProperty & "==1"
-- Label: Doing
set theRecords to search theProperty & "==2"
-- Label: Revisar
set theRecords to search theProperty & "==3"
-- Label: Done
set theRecords to search theProperty & "==4"
-- Label: Someday
set theRecords to search theProperty & "==5"
-- Label: Label
set theRecords to search theProperty & "==6"
-- Label: Delete
set theRecords to search theProperty & "==7"
```
